```python
class FrameType(IntEnum):
    AUDIO_CAE = 0x01
    VIDEO = 0x02
    AUDIO_RAW = 0x03
    CONTROL = 0x04
    TTS_AUDIO = 0x05
    # Phase 11 will ADD: MAP_IMAGE = 0x06

class MessageSchema:
    @staticmethod
    def state(state: str, robot_status: str) -> str: ...
    @staticmethod
    def transcript(text: str, speaker: str) -> str: ...
    @staticmethod
    def response(text: str) -> str: ...
    # Phase 11 will ADD: nav_status(status, destination)

@dataclass
class BinaryFrame:
    frame_type: FrameType
    payload: bytes

    @staticmethod
    def pack(frame_type: FrameType, payload: bytes) -> bytes:
        return bytes([frame_type]) + payload
```

```python
class ConnectionManager:
    def __init__(self, config: Config, event_bus: EventBus) -> None: ...

    async def send_tts_audio(self, pcm_bytes: bytes) -> None:
```
